linux  命令
	ls 		查看当前目录信息
		ls -l	以列方式显示
		ls -h	以大小单位显示，默认是字节
		ls -a	显示隐藏文件和隐藏目录
	tree 	以树状方式显示信息
	pwd		查看当前目录路径
	clear	清楚终端内容
	
	切换目录命令：
		cd 目录		切换指定目录
		cd ~		切换到当前用户主目录
		cd ..		切换到上一级目录
		cd .		切换到当前目录
		cd - 		切换到上一次目录
		
		注意 cd  后面不加 等于 cd ~
		
	创建文件
		touch 文件名		创建指定文件
		mkdir 目录名		创建目录（文件夹）
			-p		创建所依赖的文件夹
		rm 文件名或目录名	删除指定文件或者目录
			-i		交互式提示
			-r		递归删除目录及其内容
			-f		强制删除，忽略不存在的文件，无需提示
			-d		删除空目录
		rmdir 目录名		创建空目录
	
	复制、拷贝文件
		cp 文件/目录  新目录			拷贝文件、目录
			-i		交互式提示
			-r		递归拷贝目录及其内容
			-v		显示拷贝后的路径描述
			-a		保留文件的原有权限
			
		mv 文件/目录  新目录			移动文件、目录、重命名
			-i		交互式提示
			-v		显示移动后的路径描述
	重定向：
		将终端显示的内容重新放在指定文件中显示
		
		>	如果文件存在覆盖原有文件内容，相当于文件操作中的'w'模式
		>>	如果文件存在追加写入文件末尾，相当于文件操作中的'a'模式
		
	
	查看文件内容命令：
		cat		查看小型文件
		more	分屏查看大型文件
			结合管道（|）查看：
				ls /bin | more
				
	链接：
		ln -s	创建软链接
			创建软连接 要指定绝对路径
			ln -s /home/python/Desktop/AA/1.txt 1-s.txt

		ln		创建硬链接
		
	文本搜索
		grep		文本搜索
			-i	忽略大小写
			-n	显示匹配行号
			-v	显示不包含匹配文本的所有行
			^	以指定字符串开头
			$	以指定字符串结尾
			.	匹配一个非换行符的字符
			
			例：grep python 1.txt -n  grep p 1.txt -nv  grep '^c' 1.txt -ni
			可结合管道		ls | grep '^sh' -i
			
	打开文件：
		gedit  将文件以编辑器的方式开启
	
	查找文件
		find	在指定目录下查找文件/目录
			-name	根据文件名/目录名 查找
			例： find ./Desktop/AA/1*.txt
			
	压缩格式
		.gz和.bz2的压缩包需要使用tar命令来压缩和解压缩
		.zip的压缩包需要使用zip命令来压缩，使用unzip命令来解压缩
		
		tar		压缩和解压缩命令
			-c		创建打包文件
			-v		显示打包或解包的详细信息
			-f		指定文件名称，必须放到所有选项的后面
			-z		压缩或解压缩(.gz)
			-j		压缩或解压缩(.bz2)
			-x		解包
			-C		解压缩到指定目录
			
			压缩例：tar -zcvf test.tar.gz AA   tar -jcvf test.tar.bz2 AA
			解压缩：tar -zxvf test.tar.gz -C ./BB   tar -jxvf test.tar.gz -C ./BB


			
		zip		压缩成.zip格式文件
		unzip	解压缩.zip格式文件
			-d		解压缩到指定目录
			
			例：zip test*.txt   unzip test.zip -d ./BB
	
	文件权限
		chmod	修改文件权限
			u		user，表示该文件的所有者
			g		group，表示用户组
			o		other，表示其他用户
			a		all，表示所有用户
			
			+		增加权限
			-		撤销权限
			=		设置权限
			
			r		可读  4
			w		可写  2
			x		可执行  1
			-		无任何权限  0
			
			例：chmod u-r 1.txt   chmod u+r 1.txt  chmod u+x info.py 
			
	管理员权限
		sudo -s		切换root用户，获取管理员权限
		sudo		某个命令的执行需要获取管理员权限

		which python3  查找命令绝对路径
		passwd		修改当前用户密码
		who			查看有那些用户开通
			whoami  查看当前管理员用户
		ifconfig	查看ip信息
		
		shutdown -h now 立即关机

	用户管理
		useradd		创建添加用户
			-m		自动创建用户主目录，主目录的名字就是用户名
			-g		指定用户所属的用户组，默认不指定会自动创建一个同名的用户组
		
		su 		切换用户
			例：
				1. sudo useradd -m laowang
				2. sudo passwd laowang
				3. su - laowang
		
		id		查看用户组信息
			例：id laowang
			
		usermod		修改用户信息
			-G		设置一个附加组
			-g		修改用户组
			
			例：修改附加组	sudo usermod -G sudo laowang
				
		gpasswd		添加和删除附加组信息
			-a 用户名	给用户添加附加组
			-d 用户名	给用户删除附加组
			删除组：
					1. sudo gpasswd -d laowang sudo
					
		userdel		删除用户
			-r 用户名	删除用户主目录，必须要设置，否则不会删除

	用户组
		groupadd	创建(添加)用户组
		
		创建用户组使用: sudo groupadd 用户组名
		创建用户并指定用户组使用: sudo useradd -m -g 用户组 用户名
		修改用户组使用: sudo usermod -g 用户组 用户名
		删除用户组使用: sudo groupdel 用户组名

	vim操作：
		i 进入编辑模式
		Esc	进入固定模式
		：	进入末行模式
		:wq	写入保存并退出
		:x	保存退出
		:q!	强制退出
		

		yy	复制光标所在行
		p	粘贴
		dd	删除/剪切当前行
		V	按行选中
		u	撤销
		ctr+r	反撤销
		>>	往右缩进
		<<	往左缩进
		:/搜索的内容	搜索指定内容
		:%s/要替换的内容/替换后的内容/g	全局替换
		:开始行数,结束行数s/要替换的内容/替换后的内容	局部替换
		.	重复上一次命令操作
		G	回到最后一行
		gg	回到第一行
		数字+G	回到指定行
		shift+6	回到当前行的行首
		shift+4	回到当前行的行末
		ctr+f	下一屏
		ctr+b	上一屏
	
	安装软件
		在线安装：
			sudo apt-get install 安装包
		在线删除：	
			sudo apt-get remove 应用名 

		离线安装：
			sudo dpkg -i 离线文件包
			
		离线卸载：
			sudo dpkg –r 安装包名
